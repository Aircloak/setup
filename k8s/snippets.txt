# create secrets and configmaps
# TODO: delete secrets&configmaps kubectl --namespace
#!/bin/sh

kubectl create namespace aircloak || true

#### AIRDB CONFIG
# kubectl --namespace=aircloak delete secret air-db-config-secret || true
kubectl --namespace=aircloak create secret generic air-db-config-secret --from-file=../docker-compose/config/air_db/config.json

#### AIR CONFIG
kubectl --namespace=aircloak create secret generic air-config-secret --from-file=../docker-compose/config/air/config.json
kubectl --namespace=aircloak create secret generic air-starter --from-file=k8s/start_the_air.sh

### CLOAK CONFIG
kubectl --namespace=aircloak create configmap cloak-configmap --from-file=docker-compose/config/cloak --from-file=docker-compose/config/cloak/data_sources

# exclude namespace.yaml
kubectl --namespace=aircloak apply -f ./aircloak/*.yaml
